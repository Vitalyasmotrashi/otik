\section{Группоиды}
\subsection{Алгебра, группоид}
\begin{frame}{\insertsubsection}
\termin{Алгебра} "--- множество $G$ (носитель) с~заданным на нём набором операций, удовлетворяющим некоторой системе аксиом.

\termin{Группоид} "--- алгебра $\mathcal{G}=(G,\cdot)$, сигнатура которой состоит из одной бинарной операции $\cdot$: $G\times G \to G$.

% \termin{Полугруппа} "--- операция ассоциативна "--- $\forall a,\,b,\,c \in G$:
% $a\cdot (b\cdot c)= (a\cdot b)\cdot c$
\end{frame}

\subsection{Полугруппа, моноид}
\begin{frame}{\insertsubsection}
\termin{Полугруппа} "--- группоид, операция ассоциативна "--- $\forall a,\,b,\,c \in G$:
$a\cdot (b\cdot c)= (a\cdot b)\cdot c$

\termin{Моноид} "--- полугруппа c~единицей:
$\exists \bold{1}:  \forall a \in G  ~~ a\cdot\bold{1}=\bold{1}\cdot a=a$,\\$\bold{1}$ "--- нейтральный элемент (единица) моноида

\termin{Группа} "--- моноид, в~котором для каждого элемента существует обратный.

\end{frame}

\subsection{Группа}
\begin{frame}{\insertsubsection}
\begin{adjustwidth}{-0.0em}{-1.3em}
Множество $G$ с~% алгебраической 
операцией $\cdot$ %называется группой, если выполняются следующие условия:
"--- \termin{группа}, если:
\begin{enumerate}

\item операция $\cdot$ в~$G$ ассоциативна: $a\cdot(b\cdot c)=(a\cdot b)\cdot c~~ \forall a,b,c\in G$;

\item в~$G$ существует единица (нейтральный элемент) $\bold{1}\colon\, a\cdot\bold{1}=\bold{1}\cdot a=a~ \forall a\in G$;

\item для каждого $a\in G$ существует обратный: $a^{-1}\in G\colon\, a\cdot a^{-1}=a^{-1}\cdot a=\bold{1}$.
\end{enumerate}

\end{adjustwidth}
\end{frame}

\subsection{Аксиомы кольца}
\begin{frame}{\insertsubsection}
\begin{adjustwidth}{-0.0em}{-1.3em}
$\mathcal{K}= (\mathbb{K}, +, \cdot, \bold{0}, \bold{1})$, %сигнатура которой состоит из двух бинарных и~двух нульарных операций, 
причём для любых $a, b, c\in\mathbb{K}$:
\vspace{-1ex}

\begin{enumerate}
\item $a+(b+c)=(a+b)+c$;
\item $a+b=b+a$;
\item $a+\bold{0}=a$;
\item для каждого $a\in \mathbb{K}$ существует элемент $(-a)$, такой, что $a+(-a)=\bold{0}$;
\item $a\cdot (b\cdot c)=(a\cdot b)\cdot c$;
\item $a\cdot\bold{1}= \bold{1}\cdot a=a$;
\item $a\cdot (b+c)= a\cdot b+a\cdot c,~ (b+c)\cdot a=b\cdot a+c\cdot a$.
\end{enumerate}

% Операцию ${}+$ называют сложением кольца, операцию умножением кольца, элемент \bold{0} — нулем кольца, элемент \bold{1} — единицей кольца.

\end{frame}

\begin{block}{Конечное поле или поле Галуа}
Поле, состоящее из конечного числа элементов.
% Наиболее известным примером конечного поля является поле классов вычетов по простому модулю, т.е. факторкольцо \mathbb{Z}/(p), где p - простое число[1].
\mbox{$\mathbb{F}_q$ или $\GF(q)$,} где $q$ "--- число элементов (мощность).
% С точностью до изоморфизма конечное поле полностью определяется его мощностью, которая всегда является степенью какого-либо простого числа, т.е. 
\end{block}

$q=p^n$, где $p$ "--- простое число (\termin{характеристика} поля), $n \in \Naturalset$.


С~точностью до изоморфизма:

\indent\rlap{для $q=p$}\hspace*{6em}$\GF(q) = \mathbb{Z}_p$
% Расширение конечного поля GF(p^m) может быть задано разными полиномами p(x) одинаковых степеней m. Ненулевыми элементами любого поля порядка p^m является полный набор возможных многочленов степени m - 1 и ниже, отличающийся для разных полиномов p(x) лишь порядком следования элементов p по степеням примитивного элемента.

\indent\rlap{для $q=p^n$}\hspace*{6em}$\GF(q)$ 
"--- расширение поля $\mathbb{Z}_p$
% (поле вычетов многочленов по~модулю неприводимого многочлена степени $n$)
% (изоморфно полю разложения многочлена $x^q - x$ над полем $\mathbb{Z}_p$).
% Любое конечное поле из q = p^n элементов изоморфно полю разложения многочлена x^q - x над полем \mathbb{F}_p.[16]
