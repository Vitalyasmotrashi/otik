\item  Источники символов алфавита $A$ (можно прочитать строку; дискретные качественные)  $\to$ 
% \begin{enumerate}

% даже при одинаковом контексте;
\begin{itemize}
\item \termin{стационарные} "--- вероятность символа не зависит от времени/позиции: только от контекста;
% / \itemizeball{}
\item {нестационарные} "--- при сдвиге  вероятности меняются;
\end{itemize}

\vfill 

% % равновероятный $\subseteq$ стационарный без~памяти $\subseteq$ стационарный марковский % $\subseteq$ стационарный

стационарные  $\supset$ марковские (М) $\supset$ стационарные без~памяти (БП) $\ni$ равновероятный (РВ) % один для A

\vfill 
\begin{enumerate}
% \item \termin{марковский} источник (стац./нестац.) "---вероятность символа определяется состоянием;

\item \termin{марковский} источник 
% (большинство: марк. $\subset$ стац.; но иногда «нестац.\,марк.») 
% (большинство: марковский $\subset$ стационарный; \textcolor{gray}{но иногда «нестационарный марк-й»)} 
"--- \enumerateball{а} вероятность символа определяется состоянием;  
\enumerateball{б} состояние изменяется после порождения символа  
% \\
\textcolor{gray}{(новое состояние однозначно определяется предыдущим и~порождённым символом);}

% \item 
марковский источник порядка $m$ "--- 
% состояние на $i$-м шаге зависит от~состояний на $m$ предыдущих шагах: $i-1, i-2,..., i-m$

% \hfill\textcolor{gray}{обычно считается марковский $\subset$ стационарный}

\vfill 

\item 
% источник без памяти "--- состояние неизменно; 
\termin{стационарный источник без~памяти} "--- вероятность символа $a \in A$ постоянна (равна $p(a)$);

\vfill 
\item \termin{равновероятный источник} "--- вероятность символа $a \in A$ постоянна и~одинакова для всех 
% символов
%$a \in A$ 
(%
% $\frac{1}{N}$, где $|A|=N$
% $\forall a \in A:  p(a)=\frac{1}{|A|}$);
$p(a)=\frac{1}{|A|}$).
\end{enumerate}

% \textcolor{gray}{Источник без памяти может быть нестационарным! Но в этом курсе  БП только стационарные.}
