Для заданного сообщения $C$ рассчитать:


\begin{enumerate}
\item исходную длину $|C|$ сообщения в~символах 
(трёхбитных байтах) и~битах;
% , \itemizeball{} (минимальную) разрядность кода фиксированной ширины, 
% 
% \itemizeball{} общую длину $|\text{фикс}(C)|$ кода фиксированной ширины в~битах и~трёхбитных байтах;
\vfill

\item оценку $I_{\text{БП}}(C)$ количества информации по модели без памяти в битах и~трёхбитных байтах;

\vfill

\item блочные коды сообщения $C$ без учёта контекста:
\begin{itemize}
\item \textbf{команде из 3 студентов "--- коды Хаффмана и~Шеннона"--~Фано;}
\item из 4 студентов "--- коды Хаффмана, Шеннона"--~Фано и~Шеннона;
\item из 1-2 "--- Хаффмана или Шеннона"--~Фано %по~указанию \rlap{преподавателя.}
по варианту.
\end{itemize}
\end{enumerate}
