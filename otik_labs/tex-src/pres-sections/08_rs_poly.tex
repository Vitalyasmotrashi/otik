% Информационное сообщение $a = a_0, a_1 ... a_{\nu-1}$ "--- 
\termin{Информационный полином} ($\nu$ символов) степени $\nu-1$
$a(x) = a_0 + a_1x + a_2x^2 + ... + a_{\nu - 1} x^{\nu-1}.$

\termin{Порождающий полином} $g(x)$ степени $\kappa$ \mbox{($\kappa+1$ символов, обычно $g_\kappa = 1$).}

% \termin{Полиномиальный код}  есть множество всех многочленов степени $\mu-1 = \kappa+ \nu -1$ или меньше, делящихся на $g(x)$, т.\,е.  $C(x)=g(x)\cdot A(x)$

% Если многочлен $a(x)$ умножить на $x^\kappa$, то символы, составляющие сообщение, будут коэффициентами при более высоких степеях. Далее, $a(x)x^\kappa$ разделим на примитивный многочлен $g(x)$, тем самым, найдя частное от деления $p(x)$ и остаток $r(x)$. Зашифрованное информационное слово определится многочленом:

\termin{Кодовое слово} $C$ ($\mu= \nu+\kappa$ символов) степени $\mu-1$ делится~на~$g(x)$:

\itemizeball~несистематический код
$C(x) = a(x)\cdot g(x)$;

\itemizeball~\termin{систематический код} ($\nu$ информационных и~$\kappa$~проверочных символов):
% 
% http://www.codingtheory.nsu.ru/literature/codingtheory-2011.pdf
% 1 Обозначим. $r_i(x) = x^{\mu-i}\!\!\mod g(x)$
% 
% 2.
$C(x) = a(x)\cdot x^{\kappa} - r(x),$
где $r(x) = a(x)\cdot x^{\kappa}\!\!\mod g(x)$,
\hfill 
\textcolor{gray}{$\deg(r) < \deg(g) = \kappa$}
\\
\textcolor{gray}{$r(x)$ рассчитывается без деления, по табличным $x^i \!\!\mod g(x), ~ \kappa \leqslant i < \mu$}

Полученное слово 
% Получаем
$C(x)+err(x)=\widetilde{C}(x) = g(x)\cdot p(x) + r(x)$,   \mbox{$r(x) = err(x)\!\!\mod g(x)$}
"--- \termin{синдром},
% (иногда синдром "--- многочлен, не равный $r(x)$)
% при $err(x)=0$ и~$r(x) = 0$, 
% нулевой $r(x)$ означает успешную передачу,
$r(x) \neq 0$ "--- сбой\\
(но для Рида"--~Соломона синдромом называется другой многочлен).

\subsection{Циклические коды}
\begin{frame}{\insertsubsection}
\begin{adjustwidth}{-0.5em}{-1.em}
\setlength{\parskip}{0.3\parskip}
\footnotesize

"--- циклическая перестановка символов в кодовом слове дает другое допустимое слово того же кода:
% \vfill
% Пусть кодовое слово $C_2$ получено из $C_1$ путем циклического сдвига символов влево
%
$C_1 = (c_0, c_1, \ldots c_{\mu-1})$,
$C_2 = (c_{\mu-1}, c_0, c_1, \ldots c_{\mu-2})$


Таким образом, $C_2 = x\cdot C_1 - c_{\mu-1}\cdot (x^\mu - 1)$ $\implies c_{\mu-1}\cdot (x^\mu - 1) = x\cdot C_1 - C_2$.
\vfill

Полиномиальный код циклический $\Leftrightarrow$ $x^\mu - 1$ делится на $g(x)$.

\vfill

Проверочный многочлен 
% $h$:
$h(x) = \frac{x^\mu - 1}{g(x)}$
используется для извлечения информации из несистематического кода:
% ????????????????????????????????/
% $h(x) = \dfrac{x^\mu - 1}{g ( x^{-1} ) \cdot x^{\deg(g)}}$
% {$g(x)h(x)=0\mod x^\mu-1$}

 
% $C(x)h(x)=a(x)g(x)h(x)=0\mod x^\mu-1$

$C(x)h(x)=a(x)g(x)h(x)=a(x)\cdot(x^\mu - 1) = a(x)\cdot x^\mu - a(x)$

% Степень 
$\mu = \nu + \kappa > \deg(a) = \nu-1$ "--- две разнесённых копии коэф-тов $+a$ и~$-a$.

\vfill

% пример из /tex-src/pres-sections/08_rs_z5.tex — \subsection{Несистематический код Рида-Соломона}
$\diamondsuit$ символы из $\Integerset_5$, $a(x) = 3x + 1$, $\mu = 4$,
$g(x) =  x^2 + 4x + 3 \implies h(x) = \frac{x^4 - 1}{g(x)} = \frac{x^4 + 4}{g(x)} = x^2 + x + 3$


$C(x) = a(x)g(x) = 3 x^3 + 3 x^2 + 3 x + 3$

$C(x)h(x) = (3 x^3 + 3 x^2 + 3 x + 3)(x^2 + x + 3) = 3 x^5 + x^4 + 2 x + 4 = 3 x^5 + x^4 - 3 x - 1$

\subsection{Полиномиальный код Рида-Соломона}
\begin{frame}{\insertsubsection}
\begin{adjustwidth}{-0.5em}{-1.8em}
% \setlength{\parskip}{0.2\parskip}
\setlength{\parskip}{0.1\parskip}
\footnotesize

% Коды Рида"--~Соломона "--- %важный 
% частный случай БЧХ, 
Корни $g(x)$ Рида"--~Соломона 
% которого 
лежат в том же поле, над каким и строится код
% $(\kappa = 1)$.

Пусть $\beta$ "--- элемент поля \GF(q) порядка $\mu$ (обычно "--- примитивный элемент). 
% Если $\beta$ "--- примитивный элемент, то его порядок равен $q - 1$.
% , то есть $\beta^{q-1}=1,\quad \beta^i \neq 1, 0 < i<q-1$. 
% Пусть $\beta$ "--- примитивный элемент.

Тогда 
% нормированный полином $g(x)$ минимальной степени над полем \GF(q), корнями которого являются $d - 1$ подряд идущих степеней $\beta^{l_0}, \beta^{l_0+1},...,\beta^{l_0+d-1}$ элемента $\beta$, является порождающим полиномом кода Рида"--~Соломона над полем \GF(q):
порождающий полином кода Рида"--~Соломона:

$g(x) = (x - \beta^{l_0})(x - \beta^{l_0+1})\dots(x - \beta^{l_0+\kappa-1})$, \hfill $\deg(g) = \kappa$.


где $l_0$ "--- некоторое целое число.
% (в том числе 0 и 1), с помощью которого иногда удается упростить кодер. 
Обычно $l_0 = 1$.  


\vfill

Длина полученного кода $\mu$,
минимальное расстояние $\delta$,

проверочных символов $\kappa = \delta - 1 = \deg(g)$, \\ информационных символов $\nu = \mu - \kappa = \mu - \delta + 1$. 
% Таким образом $\delta = \mu - \nu - 1$ и код Рида — Соломона является разделимым кодом с максимальным расстоянием (является оптимальным в смысле границы Синглтона).
\bigskip

Если $\beta$ "--- примитивный элемент \GF(q), то $\mu = q - 1$.
% % (количество информационных символов $\nu$ максимально).
Количество проверочных $\kappa$ однозначно определяет $g(x)$.% и кодовое расстояние.

Исправляется до $\kappa/2$ ошибок.

\end{adjustwidth}
\end{frame}
